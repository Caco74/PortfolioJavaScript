              var dineroRecolectado = costo * cantParticipantes;

            function cantidadRecolectada(costoEstudiante,costoAdulto,cantEstudiantes,cantAdultos) {
              var dineroRec = (costoEstudiante*cantEstudiantes) + (costoAdulto*cantAdultos);

            function limpiarNombreParticipante(x,y) {
              var apellido = y.trim().toUpperCase();
              var nombre = x.trim().toLowerCase();
              return `${apellido}, ${nombre}`;
            }
            console.log(encabezadoEvento("Cancha","Rosario"));
            console.log(limpiarNombreParticipante("FRanco","demetrio")); // Devuelve --> Demetrio, franco

              var min = 1000;

                  min = monto;
                }
              }
              return min;
            }

            function maximo() {
              var max = 0;
              for(monto of donaciones) {
                if (monto > max) {
                  max = monto;
                }
              }
              return max;
            }

            function promedio() {
              var cant = 0;
              var suma = 0;
              for(monto of donaciones) {
                if(monto > min && monto < max) {
                  suma += monto;
                  cant++;
                }
              }
              return suma/cant;
            }

            var donaciones = [61,15,22,25,30,52,18,12,40];
            var min = minimo(donaciones);
            var max = maximo(donaciones);
            console.log(min);                            // Devuelve 12
            console.log(max);                            // Devuelve 61
            console.log(promedio(min, max, donaciones)); // Devuelve 28.857142857142858
